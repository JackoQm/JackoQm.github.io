# ls 
alias l='ls -F'
alias ll='ls -lh'
alias la='ls -FAh'
alias lla='ls -lAh'

alias rmr='rm -rf'

alias cpr='cp -r'

# git 
# 以图表形式简洁地显示 git 的 log 日志
alias glog='git log --graph --pretty=oneline --abbrev-commit'
# git 当前状态
alias gst='git status'
# git 添加未追踪文件到缓存区并提交(commit)
alias gcma='git commit -am '
